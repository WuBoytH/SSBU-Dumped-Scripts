    let boma = fighter.boma();
    frame(lua_state, 2.0);
    if is_excute(fighter) {
        ArticleModule::remove_exist(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER);
        ArticleModule::generate_article(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER, true);

        ArticleModule::remove_exist(boma, *FIGHTER_FOX_GENERATE_ARTICLE_BLASTER);
    }
}
