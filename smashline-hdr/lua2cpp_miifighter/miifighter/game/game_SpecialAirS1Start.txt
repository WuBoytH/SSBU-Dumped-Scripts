    let boma = fighter.boma();
    if is_excute(fighter) {
        WorkModule::on_flag(boma, FIGHTER_MIIFIGHTER_STATUS_WORK_ID_FLAG_100*KICK_ENABLE_LANDING_MUL_SPEED_X);
        notify_event_msc_cmd(0x2127e37c07, *GROUND_CLIFF_CHECK_KIND_NONE);
    }
    frame(lua_state, 14.0);
    if is_excute(fighter) {
        WorkModule::on_flag(boma, FIGHTER_MIIFIGHTER_STATUS_WORK_ID_FLAG_100*KICK_ENABLE_MOTION_SPEED_MUL);
    }
    frame(lua_state, 15.0);
    if is_excute(fighter) {
        FighterAreaModuleImpl::enable_fix_jostle_area(boma, 2, 3);
    }
    frame(lua_state, 16.0);
    if is_excute(fighter) {
        ATTACK(fighter, 0, 0, Hash40::new("top"), 2.0, 75, 100, 60, 0, 4.0, 0.0, 5.5, 2.5, None, None, None, 1.0, 0.0, *ATTACK_SETOFF_KIND_OFF, *ATTACK_LR_CHECK_F, false, 0, 0.0, 0, false, false, false, false, true, *COLLISION_SITUATION_MASK_GA, *COLLISION_CATEGORY_MASK_NO_FLOOR, *COLLISION_PART_MASK_ALL, false, Hash40::new("collision_attr_rush"), *ATTACK_SOUND_LEVEL_M, *COLLISION_SOUND_ATTR_KICK, *ATTACK_REGION_KNEE);
        WorkModule::on_flag(boma, FIGHTER_MIIFIGHTER_STATUS_WORK_ID_FLAG_100*KICK_HIT_CHECK_ONOFF);
    }
    frame(lua_state, 30.0);
    if is_excute(fighter) {
        notify_event_msc_cmd(0x2127e37c07, *GROUND_CLIFF_CHECK_KIND_ALWAYS_BOTH_SIDES);
    }
    frame(lua_state, 31.0);
    if is_excute(fighter) {
        WorkModule::on_flag(boma, FIGHTER_MIIFIGHTER_STATUS_WORK_ID_FLAG_100*KICK_GRAVITY_ONOFF);
        WorkModule::off_flag(boma, FIGHTER_MIIFIGHTER_STATUS_WORK_ID_FLAG_100*KICK_HIT_CHECK_ONOFF);
        AttackModule::clear_all(boma);
    }
    frame(lua_state, 35.0);
    if is_excute(fighter) {
        FighterAreaModuleImpl::enable_fix_jostle_area(boma, 4, 3);
    }
}
