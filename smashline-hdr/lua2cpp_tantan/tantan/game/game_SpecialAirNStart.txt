unsafe fn game_specialairnstart(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        ArticleModule::generate_article(boma, *FIGHTER_TANTAN_GENERATE_ARTICLE_SPIRALSIMPLE);
        ArticleModule::set_visibility_whole(boma, *FIGHTER_TANTAN_GENERATE_ARTICLE_SPIRALSIMPLE, false);
    }
    frame(lua_state, 3.0);
    if is_excute(fighter) {
        ArticleModule::set_visibility_whole(boma, *FIGHTER_TANTAN_GENERATE_ARTICLE_SPIRALSIMPLE, true);
    }
}
