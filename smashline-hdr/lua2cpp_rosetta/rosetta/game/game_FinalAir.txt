#[acmd_script( agent = "rosetta", script = "game_finalair", category = ACMD_GAME, low_priority )]
unsafe fn game_finalair(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        CHECK_VALID_FINAL_START_CAMERA(fighter, 0, 7, 20, 0, 0, 0);
        SLOW_OPPONENT(fighter, 15, 70);
    }
    frame(lua_state, 1.0);
    FT_MOTION_RATE(fighter, 0.5);
    if !WorkModule::is_flag(boma, *FIGHTER_INSTANCE_WORK_ID_FLAG_DISABLE_FINAL_START_CAMERA) {
        if is_excute(fighter) {
            FT_SET_FINAL_FEAR_FACE(fighter, 55);
        }
    }
    if sv_animcmd::get_value_float(fighter, *SO_VAR_FLOAT_LR) < 0.0 {
        if is_excute(fighter) {
            REQ_FINAL_START_CAMERA(fighter, Hash40::new("d04final02.nuanmb"), true);
        }
        else{
        if is_excute(fighter) {
            PostureModule::scale(boma, 4, 0);
            0x11e570(1469557358, 0.9);
            CAM_ZOOM_IN_arg5(0, 0);
            camera!(fighter, *MA_MSC_CMD_CAMERA_CAM_OFFSET, 0, 13);
            camera!(fighter, *MA_MSC_CMD_CAMERA_CAM_RECT, 0, 0, 50, 0);
        }
        else{
        if is_excute(fighter) {
            REQ_FINAL_START_CAMERA(fighter, Hash40::new("d04final.nuanmb"), true);
        }
        if is_excute(fighter) {
            FT_START_CUTIN(fighter);
        }
    }
}
}
frame(lua_state, 50.0);
if is_excute(fighter) {
CAM_ZOOM_OUT(fighter);
}
}
