    let boma = fighter.boma();
    if WorkModule::is_flag(boma, *FIGHTER_PACKUN_STATUS_SPECIAL_N_WORK_FLAG_END_PREVIOUS_SHOOT) {
        frame(lua_state, 1.0);

    frame(lua_state, 3.0);

        PLAY_SE(fighter, Hash40::new("se_packun_special_n05"));
    }
}
