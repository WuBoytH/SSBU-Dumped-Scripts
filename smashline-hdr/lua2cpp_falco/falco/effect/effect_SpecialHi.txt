    if is_excute(fighter) {
        BURN_COLOR(fighter, 2, 0.1, 0, 0.5);
    }
    wait(lua_state, 2.0);
    if is_excute(fighter) {
        BURN_COLOR_FRAME(fighter, 2, 1, 0.2, 0.1, 0);
    }
    wait(lua_state, 2.0);

        BURN_COLOR_NORMAL(fighter);
    }

}

}
