#[acmd_script( agent = "reflet", script = "effect_entryl" , category = ACMD_EFFECT , low_priority)]
unsafe fn effect_entryl(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        EFFECT(fighter, Hash40::new("reflet_entry"), Hash40::new("top"), 0, 0, 0, 0, 270, 0, 1, 0, 0, 0, 0, 0, 0, true);
        FLASH(fighter, 0.8, 0.8, 0.5, 0.8);
        BURN_COLOR(fighter, 2.6, 2.6, 0.85, 0.9);
        ColorBlendModule::set_disable_camera_depth_influence(boma, true);
    }
    wait(lua_state, 1.0);
    if is_excute(fighter) {
        FLASH_FRM(fighter, 35, 0.8, 0.8, 0.4, 0);
        BURN_COLOR_FRAME(fighter, 40, 2.6, 2.6, 0.85, 0);
    }
    wait(lua_state, 35.0);
    if is_excute(fighter) {
        COL_NORMAL(fighter);
        BURN_COLOR_NORMAL(fighter);
        ColorBlendModule::set_disable_camera_depth_influence(boma, false);
    }
}
