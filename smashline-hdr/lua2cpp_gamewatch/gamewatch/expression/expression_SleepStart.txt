#[acmd_script( agent = "gamewatch", script = "expression_sleepstart" , category = ACMD_EXPRESSION , low_priority)]
unsafe fn expression_sleepstart(fighter: &mut L2CAgentBase) {
	let lua_state = fighter.lua_state_agent;
	let boma = fighter.boma();
	if is_excute(fighter) {
		slope(*MA_MSC_CMD_SLOPE_SLOPE, *SLOPE_STATUS_TOP);
		VisibilityModule::set_int64(boma, Hash40::new("hand"), Hash40::new("hand_hold_lr"));
	}
	wait(lua_state, 9.0);
	if is_excute(fighter) {
		VisibilityModule::set_int64(boma, Hash40::new("hand"), Hash40::new("hand_normal"));
	}
}
