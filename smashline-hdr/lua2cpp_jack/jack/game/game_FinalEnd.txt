#[acmd_script( agent = "jack", script = "game_finalend" , category = ACMD_GAME , low_priority)]
unsafe fn game_finalend(fighter: &mut L2CAgentBase) {
	let lua_state = fighter.lua_state_agent;
	let boma = fighter.boma();
	if is_excute(fighter) {
		CANCEL_FILL_SCREEN(fighter, 0, 0);
		CANCEL_FILL_SCREEN(fighter, 1, 0);
		ATTACK_ABS(fighter, *FIGHTER_JACK_ATTACK_ABSOLUTE_KIND_FINAL, 0, 13.0, 60, 100, 0, 73, 0.0, 1.0, *ATTACK_LR_CHECK_POS, 0.0, true, Hash40::new("collision_attr_normal"), *ATTACK_SOUND_LEVEL_L, *COLLISION_SOUND_ATTR_NONE, *ATTACK_REGION_NONE);
		WorkModule::on_flag(boma, *FIGHTER_JACK_STATUS_FINAL_FLAG_ABS_SET);
	}
	frame(lua_state, 15.0);
	if is_excute(fighter) {
		WorkModule::on_flag(boma, *FIGHTER_JACK_STATUS_FINAL_FLAG_END_FINAL);
	}
	frame(lua_state, 20.0);
	if is_excute(fighter) {
		WorkModule::on_flag(boma, *FIGHTER_STATUS_VISUAL_SCENE_FLAG_END_EXIT);
	}
}
