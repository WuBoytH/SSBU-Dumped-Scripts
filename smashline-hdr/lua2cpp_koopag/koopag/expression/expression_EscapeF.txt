#[acmd_script( agent = "koopag", script = "expression_escapef", category = ACMD_EXPRESSION, low_priority )]
unsafe fn expression_escapef(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    if is_excute(fighter) {
        slope!(fighter, *MA_MSC_CMD_SLOPE_SLOPE, *SLOPE_STATUS_LR);
    }
    frame(lua_state, 11.0);
    if is_excute(fighter) {
        ItemModule::set_attach_item_visibility(boma, false);
        VisibilityModule::set_int64(boma, Hash40::new("body"), Hash40::new("body_shell"));
    }
    frame(lua_state, 37.0);
    if is_excute(fighter) {
        ItemModule::set_attach_item_visibility(boma, true);
        VisibilityModule::set_int64(boma, Hash40::new("body"), Hash40::new("body_normal"));
    }
}
