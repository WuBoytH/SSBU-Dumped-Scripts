#[acmd_script( agent = "trail", script = "game_specialairlwstart" , category = ACMD_GAME , low_priority)]
unsafe fn game_specialairlwstart(fighter: &mut L2CAgentBase) {
	let lua_state = fighter.lua_state_agent;
	let boma = fighter.boma();
	frame(lua_state, 7.0);
	if is_excute(fighter) {
		FighterAreaModuleImpl::enable_fix_jostle_area(boma, 9, 3);
	}
	frame(lua_state, 8.0);
	if is_excute(fighter) {
		WorkModule::on_flag(boma, *FIGHTER_TRAIL_STATUS_SPECIAL_LW_FLAG_SHIELD);
	}
	frame(lua_state, 24.0);
	if is_excute(fighter) {
		FighterAreaModuleImpl::enable_fix_jostle_area(boma, 4, 4);
	}
	frame(lua_state, 26.0);
	if is_excute(fighter) {
		WorkModule::off_flag(boma, *FIGHTER_TRAIL_STATUS_SPECIAL_LW_FLAG_SHIELD);
	}
}
