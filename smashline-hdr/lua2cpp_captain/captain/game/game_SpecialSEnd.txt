        AttackModule::set_add_reaction_frame(boma, 0, 3.0, false);
        AttackModule::set_add_reaction_frame(boma, 1, 3.0, false);
    }
    wait(lua_state, 1.0);

        AttackModule::clear_all(boma);
        JostleModule::set_status(boma, true);
