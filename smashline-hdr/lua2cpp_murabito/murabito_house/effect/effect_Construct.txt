#[acmd_script( agent = "murabito_house", script = "effect_construct", category = ACMD_EFFECT, low_priority)]
unsafe fn effect_construct(fighter: &mut L2CAgentBase) {
    let lua_state = fighter.lua_state_agent;
    let boma = fighter.boma();
    frame(lua_state, 70.0);
    if is_excute(fighter) {
        EFFECT_FOLLOW_RND(fighter, Hash40::new("murabito_final_house_smoke"), Hash40::new("top"), 0, 5, 0, 0, 0, 0, 1.1, 0, 1, 0, 0, 0, 0, true);
    }
    frame(lua_state, 90.0);
    if is_excute(fighter) {
        VisibilityModule::set_int64(boma, Hash40::new("step1"), Hash40::new("step1_on"));
    }
    frame(lua_state, 150.0);
    if is_excute(fighter) {
        VisibilityModule::set_int64(boma, Hash40::new("step2"), Hash40::new("step2_on"));
    }
    frame(lua_state, 200.0);
    if is_excute(fighter) {
        VisibilityModule::set_int64(boma, Hash40::new("step3"), Hash40::new("step3_on"));
    }
    frame(lua_state, 215.0);
    if is_excute(fighter) {
        EFFECT_OFF_KIND(fighter, Hash40::new("murabito_final_house_smoke"), false, false);
    }
    frame(lua_state, 220.0);
    if is_excute(fighter) {
        EFFECT(fighter, Hash40::new("murabito_final_house_completion"), Hash40::new("top"), 0, -2, -4, 0, 0, 0, 1.05, 0, 1, 0, 0, 0, 0, false);
    }
    frame(lua_state, 300.0);
    if is_excute(fighter) {
        VisibilityModule::set_whole(boma, false);
        methodlib::L2CValue::as_hash()const(Hash40::new("critical_hit"), Hash40::new("top"), 5, 5, 0, 10, 0, true);
        methodlib::L2CValue::as_hash()const();
        EffectModule::request_post_effect_line_circle(boma);
        EFFECT(fighter, 0, Hash40::new("murabito_final_house_explosion"), Hash40::new("top"), 0, 0, -5, 0, 90, 0, 1.2, 0, 1, 0, 0, 0, 0, false);
    }
    frame(lua_state, 10.0);
    if is_excute(fighter) {
        EffectModule::remove_post_effect_line(boma, 5, true);
    }
}
